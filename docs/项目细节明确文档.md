# 项目细节明确文档

## 数据流向

### **SDR -> RK3588 (射频链路)**

- **数据内容：** 原始 IQ 信号（UDP 传输）。
- **时间戳打标：** 主控程序在 `sdr.rx()` 接收到数据的瞬间，挂载 **RK3588 系统时间戳**。（“主控程序在通过 `sdr.rx()` 拿到原始信号的第一时间，立刻看一眼 RK3588 的‘表’（系统时钟），并把这个精确的时间点‘印’在这包信号的数据包上，一起发给射频组去做处理。”）
- **流转逻辑：** 主控通过**共享内存**将“信号+时间戳”传给射频组；射频组分析完后，将**识别结果+原时间戳**传回主控。（“数据传递过程：主控程序把‘带时间的信号包’放在 RK3588 的一块‘公共内存区域（共享内存）’里，并告诉射频组去那里取；射频组算完结果后，会把‘分析出的答案’连同‘原来的时间标签’一起交还给主控。”）

### **K230 -> RK3588 (视觉链路)**

- **数据内容：** 视频流 + 识别报文（RTSP/网络传输）。

- **时间对齐：** 

  * 主控提取视频流自带的 **PTS（显示时间戳）**。（“主控程序从 K230 发来的视频里，找出每张画面自带的‘拍摄时间点’（这个时间点叫 PTS，记录了这张画面的先后顺序），想象 K230 拍电影时，每张胶片边缘都印了一个“第几分几秒拍的”数字。主控程序要把这个数字找出来读一遍。”）

  - 主控计算 **系统时间与 PTS 的偏移量**，将视频时间换算为系统统一时间。（“主控程序对比一下 RK3588 的‘北京时间’和视频里的‘拍摄时间’差了多少，然后通过加减这个差值，把视频的时间全都换算成统一的‘北京时间’，比如 K230 的表显示现在是“1点”，但 RK3588 的表显示是“5点”。主控程序发现差了“4小时”，以后看到 K230 画面里的时间，直接加 4 小时，就能知道这画面对应北京时间是几点了。”）

- **流转逻辑：** 视频流由主控**直接转发**给上位机；识别报文经时间换算后发给上位机。（“数据转发规则：主控程序对‘大个头’的视频画面不拆包，直接原封不动地转发给上位机（为了快）；而对‘小个头’的识别报文（如无人机位置），则先按照刚才算的‘时差’把时间改准，再发给上位机。”）

### **RK3588 -> 上位机**

- **对齐规则：** 上位机根据**统一的系统时间戳**，将同一时刻的“视频画面”与“射频识别结果（如频率、强度）”进行同步渲染。

```html
                    ctx.lineTo(x, y);

                ctx.strokeStyle = '#d946ef';
                ctx.lineWidth = 1.5;
                ctx.stroke();

                // 瀑布流平移

                ctx.putImageData(imgData, 0, spectrumHeight + 1);
```

## 主控组目标明确

```
【SDR进程】
    采集IQ → 写共享内存

【主控进程】
    读IQ（不处理）
    ↓
    发给RF进程
    ↓
    收RF结果
    ↓
    收视觉数据
    ↓
    融合
    ↓
    WebSocket发上位机

【RF进程】
    IQ → FFT → YOLO → 循环谱 → 输出结果
```

## 数据接口定义

在 **Orange Pi 5 (RK3588)** 上，主控组需要运行两个相互协作的进程。以下是完整的接口代码规范。

### 底层 C++：网口数据搬运

它独立编译运行。通过网口拉取数据，并维持一个**循环覆盖的共享内存块**。

```C++
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <stdbool.h>
#include <string.h>
#include <signal.h>
#include <errno.h>
#include <unistd.h>
#include <fcntl.h>
#include <time.h>
#include <sys/mman.h>
#include <iio.h>

#define SHM_NAME "/sdr_shared_mem"

// ========= 参数 =========
#define IQ_COUNT   16384
#define RING_SIZE  64

// ========= 运行控制 =========
static volatile int running = 1;

void handle_sigint(int sig) {
    (void)sig;
    running = 0;
}

// ========= 时间 =========
static inline uint64_t get_time_us() {
    struct timespec ts;
    clock_gettime(CLOCK_MONOTONIC, &ts);
    return ts.tv_sec * 1000000ULL + ts.tv_nsec / 1000;
}

// ========= 控制块（算法写） =========
#pragma pack(push, 1)
typedef struct {
    uint32_t cmd_id;
    uint8_t  mode;        // 0=idle 1=scan 2=dwell
    float    freq_start;
    float    freq_end;
    float    step;
    uint32_t dwell_us;
    uint8_t  update_flag;
} ControlBlock;
#pragma pack(pop)

// ========= 数据块 =========
#pragma pack(push, 1)
typedef struct {
    uint32_t seq;

    uint64_t timestamp_us;
    uint32_t frame_id;

    float center_freq;
    uint32_t dwell_id;

    int16_t iq[IQ_COUNT * 2];

} SdrDataBlock;
#pragma pack(pop)

// ========= 共享内存 =========
typedef struct {
    ControlBlock ctrl;

    uint32_t write_idx;
    SdrDataBlock ring[RING_SIZE];

} SharedMemory;

// ========= 设置频率 =========
void set_rx_freq(struct iio_device *phy, double freq_hz) {
    struct iio_channel *lo = iio_device_find_channel(phy, "altvoltage0", true);
    if (!lo) return;

    if (iio_channel_attr_write_longlong(lo, "frequency", (long long)freq_hz) < 0) {
        fprintf(stderr, "Failed to set frequency\n");
    }
}

// ========= 主程序 =========
int main() {

    signal(SIGINT, handle_sigint);

    // ---------- 1. IIO ----------
    struct iio_context *ctx = iio_create_context_from_uri("ip:192.168.1.10");
    if (!ctx) {
        perror("iio context failed");
        return -1;
    }

    struct iio_device *phy = iio_context_find_device(ctx, "ad9361-phy");
    struct iio_device *rx  = iio_context_find_device(ctx, "cf-ad9361-lpc");

    if (!phy || !rx) {
        fprintf(stderr, "Device not found\n");
        return -1;
    }

    // ---------- 2. 通道 ----------
    struct iio_channel *chn_i = iio_device_find_channel(rx, "voltage0", false);
    struct iio_channel *chn_q = iio_device_find_channel(rx, "voltage1", false);

    if (!chn_i || !chn_q) {
        fprintf(stderr, "Channel error\n");
        return -1;
    }

    iio_channel_enable(chn_i);
    iio_channel_enable(chn_q);

    struct iio_buffer *buf = iio_device_create_buffer(rx, IQ_COUNT, false);
    if (!buf) {
        fprintf(stderr, "Buffer error\n");
        return -1;
    }

    // ---------- 3. 共享内存 ----------
    int shm_fd = shm_open(SHM_NAME, O_CREAT | O_RDWR, 0666);
    if (shm_fd < 0) {
        perror("shm_open");
        return -1;
    }

    size_t shm_size = sizeof(SharedMemory);

    if (ftruncate(shm_fd, shm_size) < 0) {
        perror("ftruncate");
        return -1;
    }

    SharedMemory *shm = mmap(NULL, shm_size,
                             PROT_READ | PROT_WRITE,
                             MAP_SHARED, shm_fd, 0);

    if (shm == MAP_FAILED) {
        perror("mmap");
        return -1;
    }

    memset(shm, 0, shm_size);

    printf("SDR采集启动（支持扫频/驻留）...\n");

    uint32_t frame_id = 0;
    uint32_t dwell_id = 0;

    double cur_freq = 2.4e9;

    // ---------- 主循环 ----------
    while (running) {

        ControlBlock ctrl = shm->ctrl;

        if (ctrl.update_flag) {
            printf("收到新控制指令 cmd_id=%u\n", ctrl.cmd_id);
            shm->ctrl.update_flag = 0;
        }

        // ===== 模式选择 =====
        if (ctrl.mode == 1) {
            // ===== 扫频 =====
            for (double f = ctrl.freq_start; f <= ctrl.freq_end; f += ctrl.step) {

                if (!running) break;

                set_rx_freq(phy, f);
                usleep(2000);  // 等待锁定

                if (iio_buffer_refill(buf) < 0) continue;

                void *src = iio_buffer_first(buf);

                uint32_t idx = shm->write_idx % RING_SIZE;
                SdrDataBlock *blk = &shm->ring[idx];

                blk->seq++;
                blk->timestamp_us = get_time_us();
                blk->frame_id = frame_id++;
                blk->center_freq = f;
                blk->dwell_id = dwell_id;

                memcpy(blk->iq, src, IQ_COUNT * 4);

                blk->seq++;

                shm->write_idx++;
            }
            dwell_id++;

        } else if (ctrl.mode == 2) {
            // ===== 驻留 =====
            cur_freq = ctrl.freq_start;
            set_rx_freq(phy, cur_freq);

            uint64_t start = get_time_us();

            while (get_time_us() - start < ctrl.dwell_us) {

                if (!running) break;

                if (iio_buffer_refill(buf) < 0) continue;

                void *src = iio_buffer_first(buf);

                uint32_t idx = shm->write_idx % RING_SIZE;
                SdrDataBlock *blk = &shm->ring[idx];

                blk->seq++;
                blk->timestamp_us = get_time_us();
                blk->frame_id = frame_id++;
                blk->center_freq = cur_freq;
                blk->dwell_id = dwell_id;

                memcpy(blk->iq, src, IQ_COUNT * 4);

                blk->seq++;

                shm->write_idx++;
            }

            dwell_id++;

        } else {
            // idle 模式
            usleep(10000);
        }
    }

    // ---------- 清理 ----------
    printf("退出...\n");

    iio_buffer_destroy(buf);
    iio_context_destroy(ctx);

    munmap(shm, shm_size);
    close(shm_fd);

    return 0;
}
```

射频组提取共享内存里的数据：

```c++
#include <stdio.h>
#include <stdlib.h>
#include <stdint.h>
#include <stdbool.h>
#include <string.h>
#include <unistd.h>
#include <fcntl.h>
#include <time.h>
#include <sys/mman.h>

#define SHM_NAME "/sdr_shared_mem"

#define IQ_COUNT   16384
#define RING_SIZE  64

// ========= 控制块 =========
#pragma pack(push, 1)
typedef struct {
    uint32_t cmd_id;
    uint8_t  mode;
    float    freq_start;
    float    freq_end;
    float    step;
    uint32_t dwell_us;
    uint8_t  update_flag;
} ControlBlock;
#pragma pack(pop)

// ========= 数据块 =========
#pragma pack(push, 1)
typedef struct {
    uint32_t seq;

    uint64_t timestamp_us;
    uint32_t frame_id;

    float center_freq;
    uint32_t dwell_id;

    int16_t iq[IQ_COUNT * 2];

} SdrDataBlock;
#pragma pack(pop)

// ========= 共享内存 =========
typedef struct {
    ControlBlock ctrl;

    uint32_t write_idx;
    SdrDataBlock ring[RING_SIZE];

} SharedMemory;


// ========= Seqlock读取 =========
bool read_block_safe(SdrDataBlock *src, SdrDataBlock *dst) {
    uint32_t s1, s2;

    do {
        s1 = src->seq;

        if (s1 & 1) continue;  // 正在写

        memcpy(dst, src, sizeof(SdrDataBlock));

        s2 = src->seq;

    } while (s1 != s2 || (s2 & 1));

    return true;
}


// ========= 主程序 =========
int main() {

    // ---------- 1. 打开共享内存 ----------
    int shm_fd = shm_open(SHM_NAME, O_RDONLY, 0666);
    if (shm_fd < 0) {
        perror("shm_open failed");
        return -1;
    }

    size_t shm_size = sizeof(SharedMemory);

    SharedMemory *shm = mmap(NULL, shm_size,
                             PROT_READ,
                             MAP_SHARED,
                             shm_fd, 0);

    if (shm == MAP_FAILED) {
        perror("mmap failed");
        return -1;
    }

    printf("算法进程已连接共享内存\n");

    // ---------- 2. 初始化 ----------
    uint32_t last_read_idx = 0;
    uint32_t last_frame_id = 0;

    SdrDataBlock local_buf;

    // ---------- 3. 主循环 ----------
    while (1) {

        uint32_t write_idx = shm->write_idx;

        // 没有新数据
        if (last_read_idx == write_idx) {
            usleep(1000);
            continue;
        }

        // 防止落后太多（RingBuffer被覆盖）
        if (write_idx - last_read_idx > RING_SIZE) {
            printf("⚠️ 数据覆盖，跳帧 %u\n", write_idx - last_read_idx);
            last_read_idx = write_idx - RING_SIZE;
        }

        // ========= 逐帧读取 =========
        while (last_read_idx < write_idx) {

            uint32_t idx = last_read_idx % RING_SIZE;
            SdrDataBlock *src = &shm->ring[idx];

            // 安全读取
            read_block_safe(src, &local_buf);

            // ========= 丢帧检测 =========
            if (last_read_idx != 0 &&
                local_buf.frame_id != last_frame_id + 1) {

                printf("⚠️ 丢帧: expect=%u got=%u\n",
                       last_frame_id + 1,
                       local_buf.frame_id);
            }

            last_frame_id = local_buf.frame_id;

            // ========= 数据消费（这里接算法） =========
            printf("frame=%u freq=%.2fMHz dwell=%u\n",
                   local_buf.frame_id,
                   local_buf.center_freq / 1e6,
                   local_buf.dwell_id);

            // 👉 在这里接你的算法：
            // process_iq(local_buf.iq);

            last_read_idx++;
        }
    }

    // ---------- 清理 ----------
    munmap(shm, shm_size);
    close(shm_fd);

    return 0;
}
```

```python
import numpy as np
import socket
import json
import asyncio
from multiprocessing import shared_memory
from fastapi import FastAPI, WebSocket

# 【接口合同 2】射频组必须实现的类模板
class RF_Algorithm_Module:
    def __init__(self):
        self.state = "SCAN" # SCAN, DETECT, AUDIT
        
    def execute(self, raw_iq):
        # 射频组以后在这里写业务逻辑
        # 主控组现在写 Mock 数据用于独立开发
        mock_spectrum = np.random.uniform(-80, -40, 512).tolist()
        return {
            "label": f"Mock_{self.state}_Result",
            "spectrum": mock_spectrum,
            "is_target": True if np.random.rand() > 0.8 else False
        }

app = FastAPI()
rf_algo = RF_Algorithm_Module() # 射频组的“插件”

# 视觉数据监听器
class VisionListener:
    def __init__(self):
        self.sock = socket.socket(socket.AF_INET, socket.SOCK_DGRAM)
        self.sock.bind(("0.0.0.0", 8080))
        self.sock.setblocking(False)
        self.last_box = None

    def get_update(self):
        try:
            data, _ = self.sock.recvfrom(2048)
            self.last_box = json.loads(data)
        except: pass
        return self.last_box

vision = VisionListener()

@app.websocket("/ws/data")
async def telemetry_stream(websocket: WebSocket):
    await websocket.accept()
    shm = shared_memory.SharedMemory(name="sdr_raw_data")
    
    while True:
        # 1. 主控从共享内存“捞”数
        raw_bytes = np.frombuffer(shm.buf, dtype=np.int16, offset=16) # 避开 Header
        
        # 2. 调度射频组算法 (输入裸数，拿到识别结果)
        # 这里主控可以根据逻辑切换 rf_algo.state
        rf_res = rf_algo.execute(raw_bytes)
        
        # 3. 调度视觉组数据 (来自 K230)
        vision_res = vision.get_update()

        # 4. 封装发送给上位机的“综合全家桶”
        payload = {
            "time": time.time(),
            "waterfall": rf_res["spectrum"],  # 频谱板块数据
            "rf_info": rf_res["label"],        # 射频识别结果
            "vision_box": vision_res,         # 视觉框坐标
            "uav_detected": rf_res["is_target"]
        }
        await websocket.send_json(payload)
        await asyncio.sleep(0.04) # 25 FPS 调度
```
